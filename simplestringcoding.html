string length using while loop

                //c langugae
                #include
                  int main()
                  {
                    int i=0;
                    char str[50];
                    printf("enter the string : ");
                    gets(str);
                  
                    while(str[++i]!='\0');
                  
                    printf("your string length is  : %d",i);
                  }
